Python Code 
from deepvoice import DeepVoice

# Extract voices from an audio file
voices = DeepVoice.extract_voices("recording.wav")

# Output: List of dictionaries with speaker info
for voice in voices:
    print(f"Speaker {voice['speaker']}: {voice['start']}s - {voice['end']}s")

Python Code 
from deepvoice import DeepVoice

# Verify if two voice samples match
result = DeepVoice.verify("voice1.wav", "voice2.wav")

# Check verification result
if result[0]['verified']:
    print(f"Match! Similarity: {result[0]['distance']:.3f}")
else:
    print("No match detected")
